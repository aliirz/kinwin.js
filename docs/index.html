KinWin.js

Lightweight TypeScript DOM manipulation library

npm install kinwin

Examples

Selectors


// ID selector
kw('#myId')

// Class selector
kw('.myClass')

// Name attribute selector
kw('@inputName')

// Tag selector
kw('=div')

DOM Manipulation


const element = kw('#demo');

// Get/Set HTML
element.html()                    // get
element.html('New content')       // set

// Attributes
element.attr('title')             // get
element.attr('title', 'Hello')    // set
element.attr({                    // multiple
    id: 'new-id',
    class: 'active'
})

// Classes
element.addClass('active')
element.removeClass('disabled')

Event Handling


kw('#myButton').on('click', (e) => {
    console.log('Clicked!', e);
});

// Method chaining
kw('.item')
    .addClass('active')
    .attr('title', 'Active Item')
    .on('click', (e) => {
        e.target.classList.toggle('clicked');
    });
